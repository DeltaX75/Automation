# 二、理论部分-重要函数：

## 2.1 1个初始化函数：

```c
void GPIO_Init(GPIO_TypeDef* GPIOx, GPIO_InitTypeDef* GPIO_InitStruct);
```

## 函数介绍	GPIO_Init()

讲解位置：第13讲 00:11:13
函数定义：stm32f4xx_gpio.c	line 202

作用：初始化一个或者多个IO口(同一组)的工作模式，输出类型，速度以及上下拉方式。<br>也就是一组IO口的4个配置寄存器。 <br>
(<u>GPIOx -> MODER, GPIOx -> OSPEEDR,GPIOx -> OTYPER,GPIOx -> PUPDR</u>)<br>
GPIOx: GPIOA~GPIOK(最多11组，也就是16×11=176个IO)

```c
typedef struct					//GPIO_InitTypeDef*的定义
{
uint32_t GPIO_Pin				//指定要初始化的端口
GPIOMode_TypeDef GPIO_Mode;		//端口模式
GPIOSpeed_TypeDef GPIO_Speed;	//速度
GPIOOType_TypeDef GPIO_OType;	//输出类型
GPIOPuPd_TypeDef GPIO_PuPd;		//上拉或者下拉
}GPIO_InitTypeDef;

```



```c
 assert_param();		//参数有效性判断
```

|       变量名        | 代表含义 |
| :-----------------: | :------: |
| GPIO_TypeDef* GPIOx | 哪一组IO |
|  GPIO_InitTypeDef*  | 哪一个IO |

e.g.设置GPIO_F9，第一个设为GPIO_F，第二个设为9

## 函数使用	GPIO_Init()

```c
GPIO_InitTypeDef GPIO_InitStructure;				//声明GPIO初始化的结构体
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOF, ENABLE);			//使能GPIOF时钟
	//GPIOF9,F10初始化设置，即定义结构体GPIO_InitStructure的值
	//如果需要定义的F9和F10参数不一致，或不属于同一组(比如一个是A一个是F)，则需要定义两个结构体分别GPIO_Init()，或定义完结构体之后修改结构体数据再次运行GPIO_Init函数
	GPIO_InitStructure.GPIO_Pin		=	GPIO_Pin_9 | GPIO_Pin_10;	//LED0和LED1对应IO口 参数一样可用逻辑或|
	GPIO_InitStructure.GPIO_Mode	=	GPIO_Mode_OUT;				//普通输出模式
	GPIO_InitStructure.GPIO_OType	=	GPIO_OType_PP;				//推挽输出
	GPIO_InitStructure.GPIO_Speed	=	GPIO_Speed_100MHz;			//100MHz
	GPIO_InitStructure.GPIO_PuPd	=	GPIO_PuPd_UP;				//上拉
	//以上初始化结构体设置完毕
	GPIO_Init(GPIOF, &GPIO_InitStructure);			//调用函数，初始化GPIOF9,F10
```

### 3.2.2 代码写入-头、源文件：

#### 头文件

使用#ifndef宏定义，只引用一次，避免重复定义，节省时间。<br>

```c
#ifndef __LED_H
	#define __LED_H
	void LED_Init();
#endif
```



#### 源文件

首先`#include "led.h"`，此外还要包含顶层头文件`stm32f4xx.h`，实际操作时包含`sys.h`即可，因为sys.h内引用了stm32f4xx.h文件。<br>

##### **1.调用时钟使能函数**

此处调用`RCC_AHB1PeriphClockCmd();`	为什么是这个函数？<br>该函数在头文件`stm32f4xx_rcc.h`第589行声明，在`stm32f4xx_rcc.c`第1526行定义。<br>

```C
void RCC_AHB1PeriphClockCmd(uint32_t RCC_AHB1Periph, FunctionalState NewState)
{
  assert_param(IS_RCC_AHB1_CLOCK_PERIPH(RCC_AHB1Periph));	//在这里go to definition of 'IS_RCC_AHB1_CLOCK_PERIPH'

  assert_param(IS_FUNCTIONAL_STATE(NewState));				//在这里go to definition of 'IS_FUNCTIONAL_STATE'
  ...
}
```

##### **2.编写IO口的初始化函数**

此处调用`GPIO_Init();`	参照理论部分2.1<br>该函数在头文件stm32f4xx_gpio.h第470行声明，在`stm32f4xx_gpio.c`第202行定义。<br>

`GPIO_Init();`包含两个形参(入口参数)，一个数值，一个结构体。<br>数值形参可以<u>直接输入</u>枚举类型GPIOF，**<u>结构体需要自己定义并赋值。</u>**<br>

* 结构体的定义方法

  需要注意，用`GPIO_InitTypeDef`定义结构体。此处的`GPIO_InitTypeDef`相当于`int`，本质上源于`typedef`<br>
  
  为什么是`GPIO_InitTypeDef`	答案：`GPIO_Init();`的第二个入口参数的参数类型就是`GPIO_InitTypeDef`<br>
  
  结构体都有哪些成员呢	还是看`GPIO_InitTypeDef`的definition(定义)<br>

```c
GPIO_InitTypeDef GPIO_InitStructure;				//声明GPIO初始化的结构体
	//GPIOF9,F10初始化设置，即定义结构体GPIO_InitStructure的值
	//如果需要定义的F9和F10参数不一致，或不属于同一组(比如一个是A一个是F)，则需要定义两个结构体分别GPIO_Init()，或定义完结构体之后修改结构体数据再次运行GPIO_Init函数
	GPIO_InitStructure.GPIO_Pin		=	GPIO_Pin_9 | GPIO_Pin_10;	//LED0和LED1对应IO口 参数一样可用逻辑或|
	GPIO_InitStructure.GPIO_Mode	=	GPIO_Mode_OUT;				//普通输出模式
	GPIO_InitStructure.GPIO_OType	=	GPIO_OType_PP;				//推挽输出
	GPIO_InitStructure.GPIO_Speed	=	GPIO_Speed_100MHz;			//100MHz
	GPIO_InitStructure.GPIO_PuPd	=	GPIO_PuPd_UP;				//上拉
	//以上初始化结构体设置完毕
	GPIO_Init(GPIOF, &GPIO_InitStructure);			//调用函数，初始化GPIOF9,F10
```

调用之后，使用GPIO_SetBits()进行操作，设置好F9和F10的高电平。

```C
//方法一：GPIO_SetBits()初始化两次。
GPIO_SetBits(GPIOF,GPIO_Pin_9);
GPIO_SetBits(GPIOF,GPIO_Pin_10);
//方法二：使用逻辑或一行搞定(注意参数需要一致才可)
GPIO_SetBits(GPIOF,GPIO_Pin_9 | GPIO_Pin_10);
```

```C
while(1)
{
	GPIO_SetBits(GPIOF,GPIO_Pin_9);		//设置高电平
	GPIO_ResetBits(GPIOF,GPIO_Pin_10);	//设置低电平
	delay_ms(500);
	GPIO_ResetBits(GPIOF,GPIO_Pin_9);	//设置低电平
	GPIO_SetBits(GPIOF,GPIO_Pin_10);	//设置高电平
    delay_ms(500);						//不要忘记再加一个delay构成完整循环体
}
```

### led.c

#### 第一种：“完整”写法

```C
#include "led.h"
#include "sys.h"

void LED_Init(void)
{
	GPIO_InitTypeDef GPIO_InitStructure;
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOF, ENABLE);

	GPIO_InitStructure.GPIO_Pin		=	GPIO_Pin_9;
	GPIO_InitStructure.GPIO_Mode	=	GPIO_Mode_OUT;
	GPIO_InitStructure.GPIO_OType	=	GPIO_OType_PP;
	GPIO_InitStructure.GPIO_Speed	=	GPIO_Speed_100MHz;
	GPIO_InitStructure.GPIO_PuPd	=	GPIO_PuPd_UP;
	
	GPIO_Init(GPIOF, &GPIO_InitStructure);
	GPIO_SetBits(GPIOF,GPIO_Pin_9);
    
    GPIO_InitStructure.GPIO_Pin		=	GPIO_Pin_10;
	GPIO_InitStructure.GPIO_Mode	=	GPIO_Mode_OUT;
	GPIO_InitStructure.GPIO_OType	=	GPIO_OType_PP;
	GPIO_InitStructure.GPIO_Speed	=	GPIO_Speed_100MHz;
	GPIO_InitStructure.GPIO_PuPd	=	GPIO_PuPd_UP;
    
    GPIO_Init(GPIOF, &GPIO_InitStructure);
	GPIO_SetBits(GPIOF,GPIO_Pin_10);
}
```



#### 第二种：“简略”写法

```C
#include "led.h"
#include "sys.h"

void LED_Init(void)
{
	GPIO_InitTypeDef GPIO_InitStructure;
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOF, ENABLE);

	GPIO_InitStructure.GPIO_Pin		=	GPIO_Pin_9 | GPIO_Pin_10;
	GPIO_InitStructure.GPIO_Mode	=	GPIO_Mode_OUT;
	GPIO_InitStructure.GPIO_OType	=	GPIO_OType_PP;
	GPIO_InitStructure.GPIO_Speed	=	GPIO_Speed_100MHz;

```



### main.c

```C
#include "sys.h"		//顶层文件是stm32f4xx.h，但sys.h包含了stm32f4xx.h
#include "led.h"
#include "delay.h"

int main(void)
{
	delay_init(168);
	LED_Init();
	while(1)
	{
		GPIO_SetBits(GPIOF,GPIO_Pin_9);
		GPIO_ResetBits(GPIOF,GPIO_Pin_10);
		delay_ms(500);
		GPIO_ResetBits(GPIOF,GPIO_Pin_9);
		GPIO_SetBits(GPIOF,GPIO_Pin_10);
        delay_ms(500);
	}
}
```
